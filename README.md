```

PATH=$PATH:/usr/local/cuda/bin
export PATH

LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/usr/local/cuda/lib:/usr/local/cuda/lib64
export LD_LIBRARY_PATH
```
